## `package.json`

Only the non-dependency fields matter here. Dependencies are installed via `scripts/install-deps.sh` to always get the latest versions.

```json
{
  "private": true,
  "type": "module",
  "scripts": {
    "dev": "vite dev",
    "build": "vite build",
    "check": "biome check .",
    "check:fix": "biome check --write ."
  }
}
```

### Installing dependencies

Run `scripts/install-deps.sh` from the project root after `bun init`. This installs all runtime and dev dependencies at their latest versions.

## `vite.config.ts`

Plugin order matters: TanStack Router first (generates route tree before React compiles), then React, then Tailwind, then Cloudflare.

```ts
import { cloudflare } from "@cloudflare/vite-plugin";
import tailwindcss from "@tailwindcss/vite";
import { tanstackRouter } from "@tanstack/router-plugin/vite";
import react from "@vitejs/plugin-react";
import { defineConfig } from "vite";

export default defineConfig({
  plugins: [
    tanstackRouter({
      target: "react",
      autoCodeSplitting: true,
      routesDirectory: "src/client/routes",
      generatedRouteTree: "src/client/routeTree.gen.ts",
    }),
    react(),
    tailwindcss(),
    cloudflare(),
  ],
});
```

## `drizzle.config.ts`

Minimal config for migration generation. No database credentials needed for `drizzle-kit generate`.

```ts
import { defineConfig } from "drizzle-kit";

export default defineConfig({
  dialect: "sqlite",
  schema: "./src/server/db/schema.ts",
  out: "./migrations",
});
```

## `wrangler.jsonc`

The `migrations_dir` must match `out` in `drizzle.config.ts`.

```jsonc
{
  "$schema": "node_modules/wrangler/config-schema.json",
  "name": "my-app",
  "compatibility_date": "2025-04-01",
  "main": "src/server/index.ts",
  "assets": {
    "directory": "./dist/client",
    "binding": "ASSETS"
  },
  "d1_databases": [
    {
      "binding": "DB",
      "database_name": "my-app-db",
      "database_id": "<your-database-id>",
      "migrations_dir": "migrations"
    }
  ],
  "triggers": {
    "crons": ["0 3 * * *"]
  }
}
```

## `worker-configuration.d.ts`

Must be kept in sync with `wrangler.jsonc`.

```ts
interface Env {
  DB: D1Database;
  ASSETS: Fetcher;
  GOOGLE_CLIENT_ID: string;
  GOOGLE_CLIENT_SECRET: string;
  OAUTH_REDIRECT_URI?: string;
}
```

```json

  "assist": { "actions": { "source": { "organizeImports": "on" } } },

    "enabled": true,
    "indentStyle": "tab"
  },
  "css": {

      "cssModules": false,
      "tailwindDirectives": true
    },
    "linter": { "enabled": true },
    "formatter": {
      "enabled": true,
      "indentStyle": "tab"
    }
  },
  "linter": {
    "enabled": true,
    "rules": { "recommended": true }
```

## `mise.toml`

```toml
[tools]
bun = "latest"

[tasks.build]
depends = ["check"]
run = "bun run build"

[tasks.dev]
run = "bun run dev"

[tasks.deploy]
depends = ["build"]
run = "bunx wrangler deploy"

[tasks.check]
depends = ["check:lint", "check:types"]

[tasks."check:lint"]
run = "bunx biome check ."

[tasks."check:types"]
run = "bunx tsc --noEmit"

[tasks.fix]
run = "bunx biome check --write ."

[tasks.tail]
run = "bunx wrangler tail"

[tasks."db:generate"]
run = "bunx drizzle-kit generate"

[tasks."db:migrate"]
usage = """
arg "<target>" {
  choices "local" "remote"
}
"""
run = "bunx wrangler d1 migrations apply my-app-db --${usage_target}"

[tasks.pre-commit]
depends = ["check"]
```

## Environment Variables

Set in `.dev.vars` for local development, in Cloudflare dashboard for production:

```
GOOGLE_CLIENT_ID=<your-google-client-id>
GOOGLE_CLIENT_SECRET=<your-google-client-secret>
```

Optional: `OAUTH_REDIRECT_URI` to override the auto-derived callback URL.
